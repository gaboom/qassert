 56     /** Setup QAssert. Overrides default options,
 57      * @param _options Supported values: <br/>
 58      * 				   - empty: sets enabled to true <br/>
 59      *                 - string: sets enabled to true and ajax to this url <br/>
 60      *                 - object: sets enabled to true then overrides options with this object <br/>
 61      * @returns options.enabled

 63      * @public

 65     $.assertSetup = function(_options) {
 66         options.enabled = true;
 67         if (typeof _options === "object") {
 68             // options given
 69             options = $.extend(true, options, _options);
 70         }
 71         if (typeof _options === "string") {
 72             // url given?
 73             options.ajax = _options;
 74         }
 75         if (typeof options.ajax != "object") {
 76             // ajax option is just an url?

 78                 type: "POST",
 79                 url: options.ajax
 80             };
 81         }

 84             window.onerror = failGlobal;
 85         }

 90     /**
 91      * Selector assertion. If disabled, no-op.
 92      *
 93      * @param message	optional message
 94      * @param sizeOrCallback    optional
 95      *     - If a number, we assert that it equals to the count of selected elements.
 96      *       $(selector).assert("We need two elements", 2)
 97      *
 98      *     - If a function, we call sizeOrCallback(), where this is set to the jQuery object of the selection,
 99      *       and we assert on the return value.
100      *       $(selector).assert("Must have class foo", function(subject) {return subject.hasClass("foo")})
101      *
102      *     - Otherwise we assert that the selection is not empty.
103      *       $(selector).assert().text("FOO")
104      *
105      * @returns this, method chaining possible.
106      * @public
107      */
108     $.fn.assert = function(message, sizeOrCallback) {
109         if (options.enabled) {
110             if (typeof sizeOrCallback === "number") {
111                 // assert on the size of the selection
112                 assert(this.length === sizeOrCallback, message, this)
113             } else if (typeof sizeOrCallback === "function") {
114                 // do callback with selection, assert on the returned value
115                 var value = ($.proxy(sizeOrCallback, this))();
116                 assert(value, message, this);
117             } else {
118                 // assert whether this contains any selected elements
119                 assert(this.length > 0, message, this);
120             }
121         }
122         return this;
123     }

375     function isEmpty(value) {
376         var type = getType(value);
377         switch (type) {
378             case "undefined":
379             case "null":
380             case "nan":
381                 return true;
382             case "number":
383                 return value == 0;
384             case "string":
385                 return value == "";
386             case "boolean":
387                 return value == false;
388             case "array":
389                 return value.length == 0;
390             case "object":
391                 for (var v in value) {
392                     if (!value.hasOwnProperty || value.hasOwnProperty(v)) {
393                         return false; // property found => not empty
394                     }
395                 }
396                 return true; // no property found => empty
397         }
398         return false; // unknown things are not empty
399     }

405     function equals(a, b) {
406         if (a == b) {
407             return true;
408         }
409         var type = getType(a);
410         switch (type) {
411             case "nan":
412                 return isNaN(b);
413             case "date":
414                 return getType(b) === "date" && a.valueOf() === b.valueOf();
415             case "number":
416                 return getType(b) === "number" && a.valueOf() === b.valueOf()
417             case "regexp":
418                 return getType(b) === "regexp" &&
419                     a.source === b.source && // the regex itself
420                     a.global === b.global && // and its modifers (gmi) ...
421                     a.ignoreCase === b.ignoreCase &&
422                     a.multiline === b.multiline;
423         }
424         return false;
425     }

427     function getType(obj) {
428         if (typeof obj === "undefined") {
429             return "undefined";
430 
431         // consider: typeof null === object
432         }
433         if (obj === null) {
434             return "null";
435         }
436 
437         var type = Object.prototype.toString.call( obj )
438           .match(/^\[object\s(.*)\]$/)[1] || '';
439 
440         switch (type) {
441             case 'Number':
442                 if (isNaN(obj)) {
443                     return "nan";
444                 } else {
445                     return "number";
446                 }
447             case 'String':
448             case 'Boolean':
449             case 'Array':
450             case 'Date':
451             case 'RegExp':
452             case 'Function':
453                 return type.toLowerCase();
454         }
455         if (typeof obj === "object") {
456             return "object";
457         }
458         return undefined;
459     }

470     /**
471      * Recursive equality testing.
472      * Extracted from QUnit.
473      *
474      * Test for equality any JavaScript type.
475      * Discussions and reference: http://philrathe.com/articles/equiv
476      * Test suites: http://philrathe.com/tests/equiv
477      * Author: Philippe Rathé <prathe@gmail.com>
478      */
479     var deepEquals = function (){
480 
481         var innerEquiv; // the real equiv function
482         var callers = []; // stack to decide between skip/abort functions
483         var parents = []; // stack to avoiding loops from circular referencing
484 
485         // Call the o related callback with the given arguments.
486         function bindCallbacks(o, callbacks, args) {
487             var prop = getType(o);
488             if (prop) {
489                 if (getType(callbacks[prop]) === "function") {
490                     return callbacks[prop].apply(callbacks, args);
491                 } else {
492                     return callbacks[prop]; // or undefined
493                 }
494             }
495         }
496 
497         var callbacks = function () {
498 
499             // for string, boolean, number and null
500             function useStrictEquality(b, a) {
501                 return a === b;
502             }
503 
504             return {
505                 "string": useStrictEquality,
506                 "boolean": useStrictEquality,
507                 "number": useStrictEquality,
508                 "null": useStrictEquality,
509                 "undefined": useStrictEquality,
510 
511                 "nan": function (b) {
512                     return isNaN(b);
513                 },
514 
515                 "date": function (b, a) {
516                     return getType(b) === "date" && a.valueOf() === b.valueOf();
517                 },
518 
519                 "regexp": function (b, a) {
520                     return getType(b) === "regexp" &&
521                         a.source === b.source && // the regex itself
522                         a.global === b.global && // and its modifers (gmi) ...
523                         a.ignoreCase === b.ignoreCase &&
524                         a.multiline === b.multiline;
525                 },
526 
527                 // - skip when the property is a method of an instance (OOP)
528                 // - abort otherwise,
529                 // initial === would have catch identical references anyway
530                 "function": function () {
531                     var caller = callers[callers.length - 1];
532                     return caller !== Object &&
533                             typeof caller !== "undefined";
534                 },
535 
536                 "array": function (b, a) {
537                     var i, j, loop;
538                     var len;
539 
540                     // b could be an object literal here
541                     if ( ! (getType(b) === "array")) {
542                         return false;
543                     }
544 
545                     len = a.length;
546                     if (len !== b.length) { // safe and faster
547                         return false;
548                     }
549 
550                     //track reference to avoid circular references
551                     parents.push(a);
552                     for (i = 0; i < len; i++) {
553                         loop = false;
554                         for(j=0;j<parents.length;j++){
555                             if(parents[j] === a[i]){
556                                 loop = true;//dont rewalk array
557                             }
558                         }
559                         if (!loop && ! innerEquiv(a[i], b[i])) {
560                             parents.pop();
561                             return false;
562                         }
563                     }
564                     parents.pop();
565                     return true;
566                 },
567 
568                 "object": function (b, a) {
569                     var i, j, loop;
570                     var eq = true; // unless we can proove it
571                     var aProperties = [], bProperties = []; // collection of strings
572 
573                     // comparing constructors is more strict than using instanceof
574                     if ( a.constructor !== b.constructor) {
575                         return false;
576                     }
577 
578                     // stack constructor before traversing properties
579                     callers.push(a.constructor);
580                     //track reference to avoid circular references
581                     parents.push(a);
582 
583                     for (i in a) { // be strict: don't ensures hasOwnProperty and go deep
584                         loop = false;
585                         for(j=0;j<parents.length;j++){
586                             if(parents[j] === a[i])
587                                 loop = true; //don't go down the same path twice
588                         }
589                         aProperties.push(i); // collect a's properties
590 
591                         if (!loop && ! innerEquiv(a[i], b[i])) {
592                             eq = false;
593                             break;
594                         }
595                     }
596 
597                     callers.pop(); // unstack, we are done
598                     parents.pop();
599 
600                     for (i in b) {
601                         bProperties.push(i); // collect b's properties
602                     }
603 
604                     // Ensures identical properties name
605                     return eq && innerEquiv(aProperties.sort(), bProperties.sort());
606                 }
607             };
608         }();
609 
610         var innerEquiv = function () { // can take multiple arguments
611             var args = Array.prototype.slice.apply(arguments);
612             if (args.length < 2) {
613                 return true; // end transition
614             }
615 
616             return (function (a, b) {
617                 if (a === b) {
618                     return true; // catch the most you can
619                 } else if (a === null || b === null || typeof a === "undefined" || typeof b === "undefined" || getType(a) !== getType(b)) {
620                     return false; // don't lose time with error prone cases
621                 } else {
622                     return bindCallbacks(a, callbacks, [b, a]);
623                 }
624 
625             // apply transition with (1..n) arguments
626             })(args[0], args[1]) && arguments.callee.apply(this, args.splice(1, args.length -1));
627         };
628 
629         return innerEquiv;
630

685     /**
686      * Main function giving a function stack trace with a forced or passed in Error
687      *
688      * @cfg {Error} e The error to create a stacktrace from (optional)
689      * @cfg {Boolean} guess If we should try to resolve the names of anonymous functions
690      * @return {Array} of Strings with functions, lines, files, and arguments where possible
691      */
692     function printStackTrace(options) {
693         var ex = (options && options.e) ? options.e : null;
694         var guess = options ? !!options.guess : true;
695 
696         var p = new printStackTrace.implementation();
697         var result = p.run(ex);
698         return (guess) ? p.guessFunctions(result) : result;
699     }

701     printStackTrace.implementation = function() {};
702 
703     printStackTrace.implementation.prototype = {
704         run: function(ex) {
705             ex = ex ||
706                 (function() {
707                     try {
708                         this.undef();
709                         return null;
710                     } catch (e) {
711                         return e;
712                     }
713                 })();
714             // Use either the stored mode, or resolve it
715             var mode = this._mode || this.mode(ex);
716             if (mode === 'other') {
717                 return this.other(arguments.callee);
718             } else {
719                 return this[mode](ex);
720             }
721         },

723         /**
724          * @return {String} mode of operation for the environment in question.
725          */
726         mode: function(e) {
727             if (e['arguments']) {
728                 return (this._mode = 'chrome');
729             } else if (typeof window !== 'undefined' && window.opera && e.stacktrace) {
730                 return (this._mode = 'opera10');
731             } else if (e.stack) {
732                 return (this._mode = 'firefox');
733             } else if (typeof window !== 'undefined' && window.opera && !('stacktrace' in e)) { //Opera 9-
734                 return (this._mode = 'opera');
735             }
736             return (this._mode = 'other');
737         },

739         /**
740          * Given a context, function name, and callback function, overwrite it so that it calls
741          * printStackTrace() first with a callback and then runs the rest of the body.
742          *
743          * @param {Object} context of execution (e.g. window)
744          * @param {String} functionName to instrument
745          * @param {Function} function to call with a stack trace on invocation
746          */
747         instrumentFunction: function(context, functionName, callback) {
748             context = context || window;
749             context['_old' + functionName] = context[functionName];
750             context[functionName] = function() {
751                 callback.call(this, printStackTrace());
752                 return context['_old' + functionName].apply(this, arguments);
753             };
754             context[functionName]._instrumented = true;
755         },

758          * Given a context and function name of a function that has been

853         stringifyArguments: function(args) {
854             for (var i = 0; i < args.length; ++i) {
855                 var arg = args[i];
856                 if (arg === undefined) {
857                     args[i] = 'undefined';
858                 } else if (arg === null) {
859                     args[i] = 'null';
860                 } else if (arg.constructor) {
861                     if (arg.constructor === Array) {
862                         if (arg.length < 3) {
863                             args[i] = '[' + this.stringifyArguments(arg) + ']';
864                         } else {
865                             args[i] = '[' + this.stringifyArguments(Array.prototype.slice.call(arg, 0, 1)) + '...' + this.stringifyArguments(Array.prototype.slice.call(arg, -1)) + ']';
866                         }
867                     } else if (arg.constructor === Object) {
868                         args[i] = '#object';
869                     } else if (arg.constructor === Function) {
870                         args[i] = '#function';
871                     } else if (arg.constructor === String) {
872                         args[i] = '"' + arg + '"';
873                     }
874                 }
875             }

971         guessFunctionNameFromLines: function(lineNo, source) {
972             var reFunctionArgNames = /function ([^(]*)\(([^)]*)\)/;
973             var reGuessFunction = /['"]?([0-9A-Za-z_]+)['"]?\s*[:=]\s*(function|eval|new Function)/;
974             // Walk backwards from the first line in the function until we find the line which
975             // matches the pattern above, which is the function definition
976             var line = "", maxLines = 10;
977             for (var i = 0; i < maxLines; ++i) {
978                 line = source[lineNo - i] + line;
979                 if (line !== undefined) {
980                     var m = reGuessFunction.exec(line);
981                     if (m && m[1]) {
982                         return m[1];
983                     } else {
984                         m = reFunctionArgNames.exec(line);
985                         if (m && m[1]) {
986                             return m[1];
987                         }
988                     }
989                 }
990             }
991             return '(?)';
